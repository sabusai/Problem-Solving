var Minimal_Angle = (h, m) => 
{
   min_angle = m*6;
  hour_angle = h*30+(15*min_angle)/180;
  angle = Math.abs(hour_angle-min_angle);

Perfect Number Check.

var Perfect_Check = (N) => {
  sum = 1;
        if(N===1){
          return "YES"
        }
        else if(N%2===1){
          return "NO"
        }
        else{
          
          for(let i=2; i*i<=N; i++){
            if(N%i===0){
              sum += i+N/i
            }
          }
        }
          return sum===N?"YES":"NO"  

};

Reverse a Number.

 var Reverse_Number = (N) => 
{
  rev = 0
  while(N>0){
  r = N%10;
  rev = rev*10+r
  N = parseInt(N/10);
  
 }
  return rev


};
